# antv-g6 示例

源码预览
查看源码
<div id="g6-chart"></div>
<script src="//gw.alipayobjects.com/os/lib/antv/g6/3.7.1/dist/g6.min.js"></script>
<script>
    const data = {
        nodes: [{
            id: 'node1',  // 第一个点
            x: 100,
            y: 200,
            label: '起始点',
            size: 60,
            labelCfg: { // 文字的配置
                position: 'center', // 文字的位置
                style: {
                    fontSize: 12,
                    fill: '#fff'
                }
            },
            style: {
                fill: '#f00', // 圆的填充色
                stroke: '#888', // 边框颜色
                lineWidth: 5, // 边框宽度
            }
        },
        {
            id: 'node2', // 第二个点
            x: 300,
            y: 200,
            label: '目标点1',
            size: 80,
        },
        {
            id: 'node3', // 第三个点
            x: 200,
            y: 300,
            label: '结束点',
            size: 80,
        },
        ], // 点集合
        edges: [{
            source: 'node1', // 对应 nodes 中的 id
            target: 'node2', // 对应 nodes 中的 id
            label: '连线1'           
        },
        {
            source: 'node2',
            target: 'node3',
            label: '连线2'
        }
        ], // 边集合
    }
    const graph = new G6.Graph({
        container: 'g6-chart',
        width: 500,
        height: 400
    }) // 完成 G6 图的初始化

    graph.data(data) // 绑定数据源
    graph.render() // 绘制矢量图 
</script>
